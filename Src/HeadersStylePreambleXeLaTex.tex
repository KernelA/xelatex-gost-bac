% Настройка стилей для section, subsection и т. д. в документе.
\usepackage[explicit]{titlesec}

% Настройка вертикальных и горизонтальных отступов.
\titlespacing*{\section}{0pt}{*4}{*4}
\titlespacing*{\subsection}{0pt}{*4}{*4}
\titlespacing*{\subsubsection}{0pt}{*4}{*4}
\titlespacing*{\paragraph}{0pt}{*4}{*4}
\titlespacing*{\subparagraph}{0pt}{*4}{*4}


% Переопределение номеров, чтобы отображалась точка в конце.
% Можно закомментировать, что в документе отображалось 1 [имя section] (после номера нет точки).
\renewcommand{\thesection}{\arabic{section}.}
\renewcommand{\thesubsection}{\arabic{section}.\arabic{subsection}.}
\renewcommand{\thesubsubsection}{\arabic{section}.\arabic{subsection}.\arabic{subsubsection}.}
\renewcommand{\theparagraph}{}
\renewcommand{\thesubparagraph}{ПРИЛОЖЕНИЕ \arabic{totappendix}.}

% Настройка форматирования section, subsection, subsubsetcion, paragraph.
\titleformat{\section}
    {\filcenter\normalsize\bfseries}
    {\thesection}
    {1em}{#1}

\titleformat{\subsection}
    {\filcenter\normalsize\bfseries}
    {\thesubsection}
    {1em}{#1}
    
\titleformat{\subsubsection}
    {\filcenter\normalsize\bfseries}
    {\thesubsubsection}
    {1em}{#1}   
    
\titleformat{\paragraph}
    {\filcenter\normalsize\bfseries}
    {}
    {1em}{#1}
    
\titleformat{\subparagraph}
    {\filleft\normalsize\bfseries}
    {\thesubparagraph}
    {1em}{#1}
